本地挂载 NFS 共享

# 创建挂载点
sudo mkdir -p /tmp/s_test

# 挂载本地共享 (使用NFSv4协议)
sudo mount -t nfs -o vers=4.1 localhost:/s_test /mnt

# 验证挂载
df -hT | grep nfs4
# 应输出类似：
# localhost:/nfs_share nfs4   503G   41G  437G    9% /mnt

# 写操作
 dd if=/dev/urandom of=/mnt/nfs_test/large_file.dat bs=1M count=10 status=progress
